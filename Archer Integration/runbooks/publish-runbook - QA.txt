#FTP
$archerCredential = Get-AutomationPSCredential -Name 'ftp_archer_cred'
$FTPUsername  = $archerCredential.UserName
$FTPPassword  = $archerCredential.GetNetworkCredential().Password
#$FTPUsername = "340026"
#$FTPPassword = "U7nxsN706r4ERD3Q"

#Storage account
$myCredential = Get-AutomationPSCredential -Name 'qa_sa_cred'
$userName = $myCredential.UserName
$password = $myCredential.GetNetworkCredential().Password
$context = New-AzureStorageContext -StorageAccountName $userName -StorageAccountKey $password


#archerentities.csv
# download the archerentities.csv file to the temp folder in runbook
Get-AzureStorageFileContent -ShareName sqm-azure-fileshare -Path "/publish/archerentities.csv" -Destination $env:temp -Context $context
$filePath = Join-Path -Path $env:temp -ChildPath 'archerentities.csv'

$RemoteFile = "ftp://ftp-eur.archer.rsa.com/SQMData/archerentities.csv"


# Create FTP Request Object
$FTPRequest = [System.Net.FtpWebRequest]::Create("$RemoteFile")
$FTPRequest = [System.Net.FtpWebRequest]$FTPRequest
$FTPRequest.Method = [System.Net.WebRequestMethods+Ftp]::UploadFile
$FTPRequest.Credentials = new-object System.Net.NetworkCredential($FTPUsername, $FTPPassword)
$FTPRequest.UseBinary = $true
$FTPRequest.UsePassive = $true
$FTPRequest.EnableSsl = $true


# Read the File for Upload
$FileContent = gc -en byte $filePath
$FTPRequest.ContentLength = $FileContent.Length

# Get Stream Request by bytes
$Run = $FTPRequest.GetRequestStream()
$Run.Write($FileContent, 0, $FileContent.Length)

# Cleanup
$Run.Close()
$Run.Dispose()
