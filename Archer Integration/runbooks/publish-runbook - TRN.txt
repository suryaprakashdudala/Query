Import-Csv $filePath | ForEach-Object {
    
    "Reading CSV file....."
    
    $username = [string]($_.userName)
    $userEmail = [string]($_.email).ToLower()

    $url = $BaseURL + "/platformapi/core/system/user/?`$filter=tolower(UserName) eq '$userEmail'"
    #$url
    $response = Invoke-RestMethod -Uri $url -Headers $header1 -ContentType "application/json" -Method POST #-Body $body
    $ExUserId = ($response.RequestedObject.Id)

    $IsAdd = "false"
    If ([string]::IsNullOrEmpty($ExUserId)) {
        
        "Entered in New user Inserting Block....."    
        $UserNameArray = $username.Split(" ")
	    [string]$UserFirstName = ""
        For ($i = 0; $i -le $UserNameArray.Length - 2; $i++) {
            $UserFirstName = $UserFirstName + " " + $UserNameArray[$i]
        }
        $UserFirstName = $UserFirstName.Trim() + " (" + $userEmail.ToLower() + ")"
        [string]$UserLastName = $UserNameArray[$UserNameArray.Length - 1]
        $UserLastName = ($UserLastName.ToString()).Trim()
        $userEmail=$userEmail.ToLower()
        $body = "{
                        `n	`"User`":
                        `n	{
                        `n		`"FirstName`":`"$UserFirstName`",
                        `n		`"LastName`":`"$UserLastName`",
                        `n      `"Username`":`"$userEmail`",
                        `n      `"email`":`"$userEmail`"
                        `n  },
                        `n  `"Contacts`": [
                        `n        {
                        `n            `"ContactType`": 7,
                        `n            `"ContactSubType`": 2,
                        `n            `"Value`": `"$userEmail`",
                        `n            `"IsDefault`": true
                        `n        }
                        `n    ],
                        `n	`"Password`":`"?Xmzc3bxsDuTSpJp`"
                        `n}
                        `n"
                
	    $headers = @{ 
                'Accept'        = 'application/json'
                'Content-Type'  = 'application/json'
                'Authorization' = 'Archer session-id="' + $SessionToken + '"'
            
            }

        $response = Invoke-RestMethod -Uri $BaseURL'/platformapi/core/system/user' -Headers $headers -ContentType 'application/json ; charset=utf-8' -Method Post -Body $body        
        #$response.RequestedObject       

        $NewUserId = $response.RequestedObject.Id
        #$NewUserId

        #----------
        if (($_.UserGroup).ToUpper() -eq "CONTROL_OPERATOR") {
            $UserGroup = $ControlOperatorGroupID  #"Control Operator"
            $IsAdd = "true"                                

            $Groupbody = '{
                "UserId":"' + $NewUserId + '",
                "GroupId":"' + $UserGroup + '",
                "IsAdd":"' + $IsAdd + '",
            }'

            $response2 = Invoke-RestMethod -Uri $BaseURL'/platformapi/core/system/usergroup' -Headers $headers -ContentType 'application/json ; charset=utf-8' -Method PUT -Body $Groupbody 
		}  

        if (($_.UserGroup).ToUpper() -eq "CONTROL_OWNER") {
            $UserGroup = $ControlOwnerGroupID  #"Control Owner"
            $IsAdd = "true"                                

            $Groupbody = '{
                "UserId":"' + $NewUserId + '",
                "GroupId":"' + $UserGroup + '",
                "IsAdd":"' + $IsAdd + '",
            }'

            $response2 = Invoke-RestMethod -Uri  $BaseURL'/platformapi/core/system/usergroup' -Headers $headers -ContentType 'application/json ; charset=utf-8' -Method PUT -Body $Groupbody 

        }          	   
    }
    else {
        #region begin - Update Existing User
        #Write-Host("Existing User Entered")

        $headers = @{ 
            'Accept'        = 'application/json'
            'Content-Type'  = 'application/json'
            'Authorization' = 'Archer session-id="' + $SessionToken + '"'
        }

        #----------
        
        if (($_.UserGroup).ToUpper() -eq "CONTROL_OPERATOR") {
            $UserGroup = $ControlOperatorGroupID  #"Control Operator"
            $IsAdd = "true"                                

            $Groupbody = '{
                "UserId":"' + $ExUserId + '",
                "GroupId":"' + $UserGroup + '",
                "IsAdd":"' + $IsAdd + '",
            }'

            $response2 = Invoke-RestMethod -Uri $BaseURL'/platformapi/core/system/usergroup' -Headers $headers -ContentType 'application/json ; charset=utf-8' -Method PUT -Body $Groupbody 
            #$response2.RequestedObject

        }  

        if (($_.UserGroup).ToUpper() -eq "CONTROL_OWNER") {
            $UserGroup = $ControlOwnerGroupID  #"Control Owner"
            $IsAdd = "true"                                

            $Groupbody = '{
                "UserId":"' + $ExUserId + '",
                "GroupId":"' + $UserGroup + '",
                "IsAdd":"' + $IsAdd + '",
            }'

            $response2 = Invoke-RestMethod -Uri $BaseURL'/platformapi/core/system/usergroup' -Headers $headers -ContentType 'application/json ; charset=utf-8' -Method PUT -Body $Groupbody 
            # $response2.RequestedObject

        }  
        
        #-----------
        #endregion              
    } 
}
